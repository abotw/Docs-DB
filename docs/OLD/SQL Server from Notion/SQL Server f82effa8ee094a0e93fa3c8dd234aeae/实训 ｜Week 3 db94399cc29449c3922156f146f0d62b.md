```sql
select *
from class
where classNo 
in (select classNo from Student where stuName like '_华%')
order by classNo desc
```

```sql
select 
departNo as '系部编号',
sum(limitnum) as '可选人数之和'
from Course
group by DepartNo
having sum(limitnum) > 100
```
